git init - создает локальный  репизоторий
(указывает системе git, что нужно отслеживать изменений в текущей папке)

git add название_файла - добавляет файлы к текущей версии

git commit - создает версию
git commit -m 'сообщение'

git log - список версий(коммитов)
git log --oneline   - сокращенная версия

git status - отображает изменение, произошедшие после последней версии

git push название_репо название_ветки  отправка локалбных изменений на удалённый репозиторий

git pull название_репозиторий  название_ветки
получение изменений из удалённого репо

git clone ссылка(ssh) - копирует удаленный репозиторий

git remote - работа с удалённым репозиторием

git remote add <название> <url> - добавление или привязка удалённого репозитория
git remote add origin git@github.com:user/project.git

git remote -v  - список удалённых репозиториев

git remote rm <название> - удаление пивязки

git remote rename <старое название> <новое название>
git remote rename origin main   #пример

git reset - откат изменений или версий

git reset - откатывает все изменения
git reset <название_файла>

git reset HEAD~кол-во  - откатывает указанное количество последних коммитов (версий)

git diff - построчное отобрежений
git diff <название_файла> - просмотр изменений в конкретном файле

git branch - просмотр имеющихся веток
git branch <название> - создание ветки
git branch -d <название> - удаление ветки

git checkout <название_ветки> - переход на другую ветку

git checkout -b <название_ветки> - создание ветки и переход на неё
